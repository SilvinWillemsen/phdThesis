    \item stability condition is wrong. Should be: 
    \begin{equation}
        h \geq \sqrt{c^2k^2 + 4\sigma_1k + \sqrt{(c^2k^2+4\sigma_1k)^2 + 16\kappa^2k^2}}
    \end{equation}
    \item Unit for membrane tension is N/m.
\end{itemize}
